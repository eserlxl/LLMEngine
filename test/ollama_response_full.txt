Here's a simple C++ function to calculate the factorial of a number using an iterative approach:

```cpp
#include <iostream>
using namespace std;

int factorial(int n) {
    int result = 1;
    for(int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

int main() {
    int num = 5;
    cout << "Factorial of " << num << " is " << factorial(num) << endl;
    return 0;
}
```
